\documentclass[tikz, border=10pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta, shapes.geometric, calc}

\begin{document}
\begin{tikzpicture}[
    gate/.style={draw, circle, minimum size=0.78cm, thick, fill=orange!20},
    op/.style={draw, circle, minimum size=0.68cm, thick, fill=gray!15},
    tnh/.style={draw, circle, minimum size=0.78cm, thick, fill=green!20},
    arr/.style={-{Stealth[length=5pt]}, thick},
    font=\small,
    lbl/.style={font=\footnotesize}
]

%% ─── CELL STATE LINE (y=4.8) ─────────────────────────────────────────
%%  C_{t-1} → ×(forget) → +(← i_t⊙C̃_t) → C_t →
\node[op, fill=orange!10] (mulf) at (2.0, 4.8) {$\times$};
\node[op]                 (addc) at (4.3, 4.8) {$+$};

\draw[arr, line width=1.6pt, blue!70!black]
    (-1.0, 4.8) --node[above,lbl,blue!70!black]{$C_{t-1}$} (mulf);
\draw[arr, line width=1.6pt, blue!70!black]
    (mulf) -- (addc);
\draw[arr, line width=1.6pt, blue!70!black]
    (addc) --node[above,lbl,blue!70!black]{$C_t$} (8.0, 4.8);

%% ─── GATES ROW (y=2.2) ───────────────────────────────────────────────
\node[gate] (fg)   at (2.0, 2.2) {$\sigma$};
\node[lbl, text=orange!80!black] at (2.0, 1.45) {forget};

\node[gate] (ig)   at (3.5, 2.2) {$\sigma$};
\node[lbl, text=orange!80!black] at (3.5, 1.45) {input};

\node[tnh]  (cand) at (4.8, 2.2) {$\tanh$};
\node[lbl, text=green!60!black]  at (4.8, 1.45) {cand.};

\node[gate] (og)   at (6.3, 2.2) {$\sigma$};
\node[lbl, text=orange!80!black] at (6.3, 1.45) {output};

%% ─── forget gate → × cell state ─────────────────────────────────────
\draw[arr] (fg) -- (mulf);

%% ─── i_t ⊙ C̃_t (y=3.5) → feeds + on cell state ─────────────────────
%%  ig and cand converge diagonally to mul_ic; result rises to addc
\node[op, fill=orange!10] (mul_ic) at (4.15, 3.5) {$\times$};
\draw[arr] (ig)     -- (mul_ic);
\draw[arr] (cand)   -- (mul_ic);
\draw[arr] (mul_ic) -- (addc);

%% ─── C_t branch → tanh(C_t) → ×(output gate) → h_t ─────────────────
%%  tanh_c branches DOWNWARD from C_t, not on the blue line itself
\coordinate (ctbr) at (6.3, 4.8);
\filldraw[black] (ctbr) circle (2pt);

\node[tnh] (tanh_c) at (6.3, 3.5) {$\tanh$};
\node[op]  (mul_o)  at (7.2, 2.8) {$\times$};

\draw[arr] (ctbr)   -- (tanh_c);
\draw[arr] (tanh_c) -- (mul_o);
\draw[arr] (og)     -- (mul_o);
\draw[arr] (mul_o)  --node[right,lbl]{$h_t$} (7.2, 1.5);

%% ─── h_{t-1} BUS (y=2.8, horizontal) ───────────────────────────────
%%  Arrives from left, distributes to all 4 gates/cand via branch dots
\draw[arr, thick] (-1.0, 2.8) --node[above,lbl]{$h_{t-1}$} (0.5, 2.8);
\filldraw[black] (0.5, 2.8) circle (1.8pt);
\draw[thick] (0.5, 2.8) -- (6.7, 2.8);

\filldraw[black] (2.0, 2.8) circle (1.8pt);  \draw[arr] (2.0, 2.8) -- (fg);
\filldraw[black] (3.5, 2.8) circle (1.8pt);  \draw[arr] (3.5, 2.8) -- (ig);
\filldraw[black] (4.8, 2.8) circle (1.8pt);  \draw[arr] (4.8, 2.8) -- (cand);
\filldraw[black] (6.3, 2.8) circle (1.8pt);  \draw[arr] (6.3, 2.8) -- (og);

%% ─── x_t BUS (x=0.5, vertical) ──────────────────────────────────────
%%  Arrives from below inside the box, joins h_{t-1} bus at (0.5, 2.8)
\draw[arr, thick] (0.5, 1.1) --node[left,lbl]{$x_t$} (0.5, 2.72);

%% ─── BOUNDING BOX ────────────────────────────────────────────────────
\draw[dashed, rounded corners=7pt, gray, thick]
    (-0.8, 0.8) rectangle (7.8, 5.5);
\node[gray, font=\small\itshape] at (3.5, 5.85) {Celda LSTM};

\end{tikzpicture}
\end{document}
